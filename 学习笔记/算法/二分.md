# 冶炼金属

找到我们想要的那个点左右两边的值的特征，蓝桥杯冶炼金属。

有些不懂，mid = right + left + 1>>1为什么要加1

```
package test;
import java.util.*;
public class jinshu {
	
	static List<Integer> A = new ArrayList<>();
	static List<Integer> B = new ArrayList<>();
	
	public static void main(String[] args) {
		
		Scanner scanner = new Scanner(System.in);
		
		int n  =  scanner.nextInt();
		
		
		for(int k=0;k<n;k++) {
			A.add(scanner.nextInt());
			B.add(scanner.nextInt());
		}
		
		long right = (long)1e9;
		long left = 1;
		long mid =0;
		
		while(right>left) {		//min
		mid =right + left>>1;   //相当于除以二向下取整
			if(check(mid, n)) {
				right = mid; 
			}else {
				left = mid +1;   //在左
			}
		}
		System.out.println(mid);
	
		
		right = (long)1e9;
		left = 1;
		while(right>left) { 		//max
			mid = right + left + 1>>1;   //??????
				if(check1(mid, n)) {
					left = mid;
				}else {
					right = mid-1;	
				}
		}
		
		System.out.println(mid);
		
	}
	
	public static boolean check(long result,int n) { //min
		for(int i=0;i<n;i++) {
			if(B.get(i)<A.get(i)/result) {//在左边
				return false;
			}
		}
		return true;
	}
	
	public static boolean check1(long result,int n) { //max
		for(int i=0;i<n;i++) {
			if(B.get(i)>A.get(i)/result) {//在you
				return false;
			}
		}
		return true;
	}
}
```
